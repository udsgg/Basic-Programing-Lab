```c++
#include <stdio.h>

struct Campaing {
    char name[50];
    float productPrice;
    float adSpend;
    int salesCount;
};

int main() {
    int N, i;

    if (scanf("%d", &N) != 1) {
        return 1;
    }

    struct Campaing campaings[N];

    for (i = 0; i < N; i++) {

        float commissionRate = 0.0;
        float totalRevenue;
        float totalCommission;
        float netProfitloss;

        if (scanf("%s %f %d %f", campaings[i].name, &campaings[i].productPrice, 
            &campaings[i].salesCount, &campaings[i].adSpend) != 4) {
            return 1;
        }

        totalRevenue = campaings[i].salesCount * campaings[i].productPrice;

        if (campaings[i].salesCount >= 20) {
            commissionRate += 0.20; 
        } else if (campaings[i].salesCount >= 10){
            commissionRate += 0.15;
        } else {
            commissionRate = 0.10;
        }

        totalCommission = totalRevenue * commissionRate;

        netProfitloss = totalCommission - campaings[i].adSpend;

        printf("--- Campaign: %s ---\n", campaings[i].name);
        printf("Sales: %d, Ad Spend: %.2f\n", campaings[i].salesCount, netProfitloss);
        printf("Rate Used: %.0f%%\n", commissionRate * 100);
        printf("Calculation: (%.2f * %.0f%%) - %.2f = %.2f\n", totalRevenue, commissionRate * 100, campaings[i].adSpend, netProfitloss);
        printf("Net Result: %.2f\n", netProfitloss);
    }
    return 0;
}
```
