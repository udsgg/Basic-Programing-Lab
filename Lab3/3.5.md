```c++
#include <stdio.h>

struct Order {
    int itemId;
    float unitprice;
    int quantity;
};

int main() {
    int N, i;
    float grandTotal = 0.0;
    float SHOPPING_FEE = 50.0;
    
    if (scanf("%d", &N) != 1) {
        return 1;
    }

    struct Order orders[N];

    for (i = 0; i < N; i++) {
        float itemTotal;
        int itemId, quantity;
        float SHOPPING_FEE;

        if (scanf("%d %f %d", &orders[i].itemId, &orders[i].unitprice, &orders[i].quantity) != 3) {
            return 1;
        }


        itemTotal = (orders[i].unitprice * orders[i].quantity) + SHOPPING_FEE;

        if (itemTotal >= 500.0) {
            itemTotal *= 0.90; 
        }

        grandTotal += itemTotal;
    }

    printf("Grand Total: %.2f\n", grandTotal);
    return 0;
    
}
```
