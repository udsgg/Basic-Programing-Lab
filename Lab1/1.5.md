#include <stdio.h>

int main() {
    int num , n , even_sum = 0, odd_sum = 0;
    if (scanf("%d", &n) != 1) {
        return 1;
    }
    while (num != 0) {
        if (scanf("%d", &num) != 1) {
            break;
        }
        for (int i = 0; i < n; i++) {
        if (num % 2 == 0) {
            even_sum += num;
        }
        else {
            odd_sum += num;
        }
    }
}
    printf("Even sum: %d \n", even_sum);
    printf("Odd sum: %d \n", odd_sum);
    return 0;
}
