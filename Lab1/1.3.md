#include <stdio.h>

void calculate(int number1, int operater, int number2);

int main() {
    int number1, number2, operater;
    
    if (scanf("%d %d %d", &number1, &operater, &number2) != 3) {
        return 1;
    }

    calculate(number1, operater, number2 );
    return 0;
}

void calculate(int number1, int operater, int number2) {
    int result;

    if (operater == 1) {
        result = number1 + number2;
        printf("%d\n", result);
    } 
    else if (operater == 2) { 
        result = number1 - number2;
        printf("%d\n", result);
    } 
    else if (operater == 3) { 
        result = number1 * number2;
        printf("%d\n", result);
    } 
    else if (operater == 4) {
        if (number2 == 0) {
            printf("Error\n");
            return;
        }
        result = number1 / number2;
        printf("%2.f\n", result);
    }
    else {
        printf("Invalid operator\n");
    }
}
