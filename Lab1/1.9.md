int main() {
    int zoneCode;
    float weight_kg;
    float totalcost = 0.0;

    if (scanf("%d %f", &zoneCode, &weight_kg) != 2) {
        printf("Invalid Input\n");
        return 1;
    }
    if (totalcost < 0.0 || zoneCode < 1 || zoneCode > 3) {
        printf("Invalid Zone Code\n");
        return 1;
    }
    else if (totalcost > 0.0 || zoneCode >= 1 || zoneCode <= 3) {
        calculater(zoneCode, weight_kg, &totalcost);
        printf("%.2f\n", totalcost);
    }
}

void calculater(int zoneCode, float weight_kg, float *totalcost) {
    if (zoneCode == 1) {
        if (weight_kg <= 5.0) {
            *totalcost = 50;
        }
        else if (weight_kg > 5.0 ) {
            *totalcost = 80;
        }
        return;
    }
    if (zoneCode == 2) {
        if (weight_kg <= 10.0) {
            *totalcost = 150;
        }
        else if (weight_kg > 10.0 ) {
            *totalcost = 250;
        }
        return;
    }
    if (zoneCode == 3) {
        *totalcost = 500.0;
    }
    return;
}
