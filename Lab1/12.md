```c++
#include <stdio.h>

void promotion(int planCode, float *dataUsage_GB, float *totalBill);

int main() {
    int planCode;
    float dataUsage_GB;
    float totalBill = 0.0;

    if (scanf("%d %f", &planCode, &dataUsage_GB) != 2) {
        return 1;
    }
    if (planCode == 1 || planCode == 2) {
        promotion(planCode, &dataUsage_GB, &totalBill);
        printf("%2.f\n", totalBill);
    } else {
        printf("Invalid Plan Code\n");
    }
    return 0;
}

void promotion(int planCode, float *dataUsage_GB, float *totalBill) {
    if (planCode == 1) {
        if (*dataUsage_GB <= 10) {
            *totalBill = 299;
            return;
        }
        else if (*dataUsage_GB > 10) {
            *totalBill = 299 + 10*(*dataUsage_GB - 10);
            return;
        }   
    }
    if (planCode == 2) {
        if (*dataUsage_GB <= 20) {
            *totalBill = 599;
            return;
        }
        else if (*dataUsage_GB > 20) {
            *totalBill = 599 + 10 * (*dataUsage_GB - 20);
            return;
        }
    }
}
```
