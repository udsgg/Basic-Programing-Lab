#include <stdio.h>

void calculate(int price_before_vat, int categoryCode, float *totalbills, float *vatamout);

int main() {
    int categoryCode;
    float price_before_vat;
    float totalBill = 0.0;
    float vatamout = 0.0;

    if(scanf("%f %d", &price_before_vat, &categoryCode) != 2) {
        return 1;
    }
    calculate(price_before_vat, categoryCode, &totalBill, &vatamout);
    if (categoryCode >= 1 && categoryCode <= 3) {
        vatamout = totalBill - price_before_vat;
        printf("VAT Amount : %.2f\n", vatamout);
        printf("Total Price : %.2f\n", totalBill);
    } else {
        printf("Invalid Category\n");
    }
    return 0;
}

void calculate(int price_before_vat, int categoryCode, float *totalbills, float *vatamout) {
    if (categoryCode == 1) {
        *vatamout = price_before_vat / 7;
        *totalbills =  price_before_vat * 1.07;
        return;
    }
    if (categoryCode == 2) {
        *vatamout = price_before_vat * 0;
        *totalbills =  price_before_vat * 1;
        return;
    }
    if (categoryCode == 3) {
        *vatamout = price_before_vat / 15;
        *totalbills =  price_before_vat * 1.15;
        return;
    }
}
