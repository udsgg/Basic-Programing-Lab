```C++
#include <stdio.h>
void while_sum(int Number, int &Sum, int &NumCount);

int Main() {
    int Number, Sum = 0, NumCount = 0;
    if (scanf("%d", &Number) != 1) {
        return 1;
    }
    while_sum(Number, Sum, NumCount);
}

void while_sum(int Number, int &Sum, int &NumCount) {
    while (Number != 0) {
        Sum += Number;
        NumCount++;
        if (scanf("%d", &Number) != 1) {
            return;
        }
    }
    if (NumCount == 0) {
        printf("No numbers entered\n");
    } else {
        double Average = static_cast<double>(Sum) / NumCount;
        printf("Sum: %d\n", Sum);
        printf("Average: %.2f\n", Average);
    }
}
```
