```C++
#include <stdio.h>

int main() {
    int num;
    do {
        printf("Enter a number (1-10):");
        if (scanf("%d", &num) != 1) {
            break;
        }
        if (num < 1 || num > 10) {
            printf("Error: Value must be 1-10.\n");
        }
    }
    while (num < 1 || num > 10);
    if (num >= 1 && num <= 10) {
        printf("input accepted: %d\n", num);
        return 0;
    }
    return 0;
}
```
