    while(daycode >= 1 || daycode <= 7) {
        if (daycode == 6 || daycode == 7) {
        printf("Weekend Relax Mode");
        return;
        }
        if(hour <= 8 || hour >= 17) {

        }

        printf("System Running (WorkDay)");

        }
    }

}
