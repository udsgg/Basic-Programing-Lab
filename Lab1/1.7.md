void calculateBill(int type, float Consumption, float *Bills) {
    if (type == 1) {
        if (Consumption <= 100) {
            *Bills = Consumption * 3.0;
        }
        else {
            *Bills = Consumption * 4.0;
        }
    }
    else if (type == 2) {
        if (Consumption <= 500) {
            *Bills = Consumption * 5.0;
        }
        else {
            *Bills = Consumption * 6.5;
        }
    }
    else {
        printf("Invalid Customer Type\n");
    }
}
