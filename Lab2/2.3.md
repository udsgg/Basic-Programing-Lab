```c++
#include <stdio.h>

int main() {
    int termCode;
    float investmentAmout;
    float aprRate = 0.0;
    float totalInterest = 0.0;

    if (scanf("%d %f", &termCode, &investmentAmout) != 2) {
        return 1;
    }
    if (termCode >= 1 && termCode <= 3) {
        if (termCode == 1) {
            if (investmentAmout < 5000) {
                aprRate = 3.0;
            } else {
                aprRate = 4.0;
            }
        }
        else if (termCode == 2) {
            if (investmentAmout < 10000) {
                aprRate = 5.0;
            } else {
                aprRate = 6.0;
            }
        }
        else if (termCode == 3) {
            aprRate = 8.0;
        }
        totalInterest = investmentAmout * aprRate;
        printf("%.2f\n", totalInterest);
    } else {
        printf("Invalid Term Code\n");
    }
    return 0;
}
```
