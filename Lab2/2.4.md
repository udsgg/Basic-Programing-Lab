```c++
#include <stdio.h>

int main() {
    float principal, monthlyPayment, interest;
    int monthCount = 0;
    const float INTERST_RATE = 0.01;
    const float PENALTY = 10.0;

    if (scanf("%f %f", &principal, &monthlyPayment) != 2) {
        return 1;
    }
    while (principal > 0) {
        interest = principal * INTERST_RATE;
        if (interest < PENALTY) {
            interest = PENALTY;
        }
        principal = principal + interest - monthlyPayment;
        monthCount++;
        printf("month %d: Remaining %.2f\n", monthCount, principal);
    }
    printf("Loan settled in %d months.\n", monthCount);
    return 0;
}
```
