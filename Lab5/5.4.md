#include <stdio.h>

#define NUM_STUDENTS 3

struct Student {
    int id;
    int scores;
}; 

int find_max_score_id(struct Student students[], int size);

int main() {
    struct Student students[NUM_STUDENTS];
    int top_student_id;
    int i;
    
    printf ("Enter deta for %d students:\n", NUM_STUDENTS);
    for (int i = 0; i < NUM_STUDENTS; i++) {
        printf("----Student %d----\n", i + 1);
        printf("Enter ID: ");
        scanf("%d", &students[i].id);   
        printf("Enter Scores: ");
        scanf("%d", &students[i].scores);
    }

    top_student_id = find_max_score_id(students, NUM_STUDENTS);

    printf("\n--- STUDENT SCORE REPORT ---\n");
    printf("ID   | Score\n");
    printf("-----|-------\n");
    for (i = 0; i < NUM_STUDENTS; i++) {
        printf("%-4d | %d\n", students[i].id, students[i].scores);
    }
    printf("\nTop Student ID: %d\n", top_student_id);

    return 0;
}

int find_max_score_id(struct Student students[], int size) {
    int max_score = students[0].scores;
    int max_id = students[0].id;
    
    for (int i = 1; i < size; i++) {
        if (students[i].scores > max_score) {
            max_score = students[i].scores;
            max_id = students[i].id;
        }
    }
    return max_id;
}
