```c++
#include <stdio.h>

int main() {
    int i;
    int DAYS = 7;
    int daily_temp[DAYS];
    int max_temp;
    
    for (i = 0; i < DAYS; i++) {
        printf("Enter temperature for Day %d ", i + 1);
        scanf("%d", &daily_temp[i]);
        if (i == 0) {
            max_temp = daily_temp[i];
        } else if (daily_temp[i] > max_temp) {
            max_temp = daily_temp[i];
        }
    }

    printf("\n---DAILY TEMPERATURE REPORT---\n");
    printf("Recorded Temperature (C): ");
    for (i = 0; i < DAYS; i++) {
        printf("%d ", daily_temp[i]);
    }
    printf("\n");
    printf("Maximum Temperature Found: %d C\n", max_temp);
    if (max_temp >= 35) {
        printf("Weathers is HOT!");
    } else {
        printf("Weathers is MODERATE");
    }
    return 0;
}
```
