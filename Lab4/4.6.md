```c++
#include <stdio.h>

int main() {
    int i;
    int STUDENTS = 5;
    int scores[STUDENTS];
    int sum = 0;
    float average = 0.0;
    int above_average_count = 0; 
    
    for (i = 0; i < STUDENTS; i++) {
        printf("Enter score for student %d: \n", i + 1);
        scanf("%d", &scores[i]);
        sum += scores[i];
    }
    
    average = (float)sum / STUDENTS;

    printf("\n---STUDENT SCORE ANALYSIS REPORT---\n");
    printf("Records Scores: ");
    for (i = 0; i < STUDENTS; i++) {
        printf("%d ", scores[i]);
        if (scores[i] > average) {
            above_average_count++;
        }
    }
    printf("\n");
    printf("Total Sum: %d \n", sum);
    printf("Average Score: %.2f \n", average);
    printf("Number of Socres Above Average: %d ", above_average_count);
    return 0;
}
```
