```c++
#include <stdio.h>

int main() {
    int ROWS = 2;
    int COLS = 4;
    int i, j;
    int data[ROWS][COLS];
    int total_sum = 0;

    printf("Enter %d integer numbers for the 2x4 matrix:\n", ROWS * COLS);
    for (i = 0; i < ROWS; i++) {
        for (j = 0; j < COLS; j++) {
            printf("Enter element for Row %d, Col %d: ", i, j);
            scanf("%d", &data[i][j]);
        }
    }

    printf("\n--- 2D ARRAY REPORT ---\n");
    printf("Recorded Matrix (2x4):\n");
    for (i = 0; i < ROWS; i++) {
        for (j = 0; j < COLS; j++) {
            printf("%4d ", data[i][j]);
            total_sum += data[i][j];
        }
        printf("\n");
    }

    printf("\nTotal Sum of all elements: %d\n", total_sum);
    return 0;
}
```
