```c++
#include <stdio.h>

int main() {
    int ROWS, COLS;
    int i, j;
    int data[ROWS][COLS];

    printf("Enter number of rows and columns: ");
    scanf("%d %d", &ROWS, &COLS);

    printf("Enter %d integer numbers:\n", ROWS * COLS);
    for (i = 0; i < ROWS; i++) {
        for (j = 0; j < COLS; j++) {
            printf("Enter element for Row %d, Col %d: ", i, j);
            scanf("%d", &data[i][j]);
        }
    }

    printf("\n--- TWO-DIMENSIONAL ARRAY REPORT ---\n");
    printf("Array Size: %d Rows x %d Columns\n", ROWS, COLS);
    printf("Content (Table Format):\n");
    for (i = 0; i < ROWS; i++) {
        for (j = 0; j < COLS; j++) {
            printf("%4d ", data[i][j]);
        }
        printf("\n");
    }

    return 0;
}
```
