```c++
#include <stdio.h>

int main() {
    int N,i;
    int passSum = 0;
    int failcount = 0;

    if (scanf("%d", &N) != 1) {
        return 1;
    }

    int scores[N];
    
    for (i = 0; i < N; i++) {
        if (scanf("%d", &scores[i]) != 1) {
        return 1;
        }

        if (scores[i] >= 50) {
            passSum += scores[i];
        } else {
            failcount++;
        }
    }

    printf("Passing Scores Sum: %d\n", passSum);
    printf("Failing Students Count: %d\n", failcount);
    return 0;
}
```
